#bin/bash
pkg install git php -y
git clone https://github.com/bagusuntungsaputra/mikmon.git
cat >start-mikhmon <<EOL
#bin/bash
if pgrep "php" >/dev/null 2>&1 ; then
killall php
fi
cd mikmon
php -S 0.0.0.0:8080
EOL
cat >update-mikhmon <<EOL
#bin/bash
mkdir ~/backup-mikhmon
cp ~/mikmon/include/config.php ~/backup-mikhmon/config.php
cp -rf ~/mikmon/voucher/. ~/backup-mikhmon/voucher/
cp -rf ~/mikmon/img/. ~/backup-mikhmon/img/
rm -rf mikmon
rm ~/backup-mikhmon/voucher/print.php
rm ~/backup-mikhmon/voucher/printbt.php
git clone https://github.com/laksa19/mikmon.git
cp ~/backup-mikhmon/config.php ~/mikmon/include/config.php
cp -rf ~/backup-mikhmon/voucher/. ~/mikmon/voucher/
cp -rf ~/backup-mikhmon/img/. ~/mikmon/img/
rm -rf ~/backup-mikhmon
./start-mikhmon
EOL
cat >remove-mikhmon <<EOL
#bin/bash
rm -rf mikmon
EOL
cat >.bashrc <<EOL	
~/start-mikhmon	
EOL
chmod +x start-mikhmon
chmod +x update-mikhmon
chmod +x remove-mikhmon
./start-mikhmon
